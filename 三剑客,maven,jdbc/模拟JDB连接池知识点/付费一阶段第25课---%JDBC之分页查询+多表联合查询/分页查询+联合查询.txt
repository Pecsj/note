	2.模糊查询
		通过like关键字   % %
		方式一(正常写SQL 赋值时候处理)
		select * from xxx where xxx like ? ;
		需要利用预处理对象赋值时候进行处理 %
		pstat.setString(1,"%"+letter+"%");
		方式二(将%放置在SQL语句里)
		"    select * from xxx where xxx like \"%\" ? \"%\"     ";
		直接赋值就可以啦
		pstat.setString(1,letter);
